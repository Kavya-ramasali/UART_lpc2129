                                                                                                                                                                                    #ifndef __LPC2148_REGACC_H__
#define __LPC2148_REGACC_H__

#include <stdint.h>

/* UART0 Register Definitions */

#define U0RBR (*(volatile uint8_t *)0xE000C000)
#define U0THR (*(volatile uint8_t *)0xE000C000)
#define U0DLL (*(volatile uint8_t *)0xE000C000)
#define U0DLM (*(volatile uint8_t *)0xE000C004)
#define U0LCR (*(volatile uint8_t *)0xE000C00C)
#define U0LSR (*(volatile uint8_t *)0xE000C014)

/* LSR masks */
#define U0LSR_THRE_MASK 0x20
#define U0LSR_BI_MASK   0x10
#define U0LSR_FE_MASK   0x08
#define U0LSR_PE_MASK   0x04
#define U0LSR_OE_MASK   0x02
#define U0LSR_DR_MASK   0x01

/* LCR masks */
#define U0LCR_DLAB_MASK      0x80
#define U0LCR_PARITY_MASK    0x38
#define U0LCR_STOP_BIT_MASK  0x04
#define U0LCR_DATA_LEN_MASK  0x03

/* LSR bit positions */
#define U0LSR_THRE_LPOS 5
#define U0LSR_BI_LPOS   4
#define U0LSR_FE_LPOS   3
#define U0LSR_PE_LPOS   2
#define U0LSR_OE_LPOS   1
#define U0LSR_DR_LPOS   0

/* LCR bit positions */
#define U0LCR_DLAB_LPOS     7
#define U0LCR_PARITY_LPOS   3
#define U0LCR_STOP_BIT_LPOS 2
#define U0LCR_DATA_LEN_LPOS 0

/* LSR Read Macros */

#define LSR_THRE_RD(val) do { \
    uint8_t tval = U0LSR; \
    val = (tval & U0LSR_THRE_MASK) >> U0LSR_THRE_LPOS; \
} while(0)

#define LSR_FE_RD(val) do { \
    uint8_t tval = U0LSR; \
    val = (tval & U0LSR_FE_MASK) >> U0LSR_FE_LPOS; \
} while(0)

#define LSR_PE_RD(val) do { \
    uint8_t tval = U0LSR; \
    val = (tval & U0LSR_PE_MASK) >> U0LSR_PE_LPOS; \
} while(0)

#define LSR_OE_RD(val) do { \
    uint8_t tval = U0LSR; \
    val = (tval & U0LSR_OE_MASK) >> U0LSR_OE_LPOS; \
} while(0)

#define LSR_DR_RD(val) do { \
    uint8_t tval = U0LSR; \
    val = (tval & U0LSR_DR_MASK) >> U0LSR_DR_LPOS; \
} while(0)

/* LCR Write Macros */

#define LCR_DLAB_WR(val) do { \
    uint8_t tval = U0LCR; \
    tval = (tval & ~U0LCR_DLAB_MASK) | ((val << U0LCR_DLAB_LPOS) & U0LCR_DLAB_MASK); \
    U0LCR = tval; \
} while(0)

#define LCR_PARITY_WR(val) do { \
    uint8_t tval = U0LCR; \
    tval = (tval & ~U0LCR_PARITY_MASK) | ((val << U0LCR_PARITY_LPOS) & U0LCR_PARITY_MASK); \
    U0LCR = tval; \
} while(0)

#define LCR_STOP_BIT_WR(val) do { \
    uint8_t tval = U0LCR; \
    tval = (tval & ~U0LCR_STOP_BIT_MASK) | ((val << U0LCR_STOP_BIT_LPOS) & U0LCR_STOP_BIT_MASK); \
    U0LCR = tval; \
} while(0)

#define LCR_DATA_LEN_WR(val) do { \
    uint8_t tval = U0LCR; \
    tval = (tval & ~U0LCR_DATA_LEN_MASK) | ((val << U0LCR_DATA_LEN_LPOS) & U0LCR_DATA_LEN_MASK); \
    U0LCR = tval; \
} while(0)

#endif
