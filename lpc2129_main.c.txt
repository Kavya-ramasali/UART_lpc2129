#include <stdint.h>
#include "lpc2129_uart.h"

int main(void)
{
    uint8_t tx_char = '3';
    uint8_t rx_char;
    uint8_t error_flags = 0;

    /* Initialize UART */
    UART0_init();

    /* Set baudrate to 9600 */
    configure_baudrate(9600,15000000);

    /* Transmit one character */
    transmit_data(tx_char);

    /* Receive the character */
    rx_char = receive_data(&error_flags);

    /* If no error, send received char to serial monitor */
    if(error_flags == 0)
    {
        transmit_data(rx_char);
    }

    while(1);

    return 0;
}
